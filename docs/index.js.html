/** 
возвращает значение площади треугольника со сторонами a,b,c
@param {number} a - first side.
@param {number} b - second side.
@param {number} c - third side.
*/

function ploshad (a,b,c) {
  let p = (a+b+c)/2;
  let pl = Math.sqrt(p*(p-a)*(p-b)*(p-c));
  return pl;
}
